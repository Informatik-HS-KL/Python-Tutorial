Die Funktionen können wir folgt erzeugt werden:
\begin{lstlisting}
>>> x, y, z = symbols('x y z')
>>> f = x**5 + y**2 + 3
>>> g = x**3 * y**2 * z + 4
>>> h = x**-3
\end{lstlisting}

\begin{enumerate}
	\item Die dritte Ableitung nach x von f kann wie folgt berechnet werden:
\begin{lstlisting}[mathescape]
>>> diff(f, x, 3)
$60x^2$
\end{lstlisting}
	\item
\begin{lstlisting}[mathescape]
>>> diff(g, x, 3, y, 2, z)
12
\end{lstlisting}
	\item Integriert wird mit \lstinline!integrate!.
\begin{lstlisting}[mathescape]
>>> integrate(f, (x, 0, 6))
$6y^2+7794$
\end{lstlisting}
	\item Der Grenzwert wird mit \lstinline!limit! berechnet.
\begin{lstlisting}[mathescape]
>>> limit(h, x, 0, '+')
$\infty$
>>> limit(h, x, 0, '-')
$-\infty$
\end{lstlisting}
\end{enumerate}
