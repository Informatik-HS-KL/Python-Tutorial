\subsection{Hello World(s)}

\begin{lstlisting}
>>> import numpy as np
>>>
>>> a = np.array([1,2,3])
>>> b = np.array([4,5,6])
>>>
>>> print(a + 1)
[2 3 4]
>>> print(a + b)
[5 7 9]
>>> print(a * 2)
[10 14 18]
>>> print(a * b)
[ 4 10 18]
>>>
>>>
>>> print(np.pi)
3.141592653589793
>>> print(np.e)
2.718281828459045
>>> print(np.euler_gamma)
0.5772156649015329
>>> print(np.PINF)
inf
>>> print(np.NINF)
-inf
>>> print(np.NAN)
nan
>>> print(np.PZERO)
0.0
>>> print(np.NZERO)
-0.0
\end{lstlisting}
